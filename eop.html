⚡ Event-Oriented Programming
What is EOP?
Event-Oriented Programming is a paradigm where the flow of the program is determined by events — user actions, messages, or signals — rather than sequential top-down execution. A central event loop listens for events and dispatches them to registered handlers.
addEventListener — The Core API
JavaScript
// Syntax: target.addEventListener(type, handler, options)
const btn = document.querySelector('#myBtn');

// Named handler — can be removed later
function handleClick(event) {
  console.log('clicked!', event.target);
}

btn.addEventListener('click', handleClick);

// Remove when no longer needed (memory leak prevention)
btn.removeEventListener('click', handleClick);

// Options object
btn.addEventListener('click', handleClick, {
  once:    true,    // auto-remove after first fire
  passive: true,    // hint: won't call preventDefault (scroll perf)
  capture: false,   // use bubble phase (default)
});

// Arrow function handler (can't be removed by ref)
btn.addEventListener('click', (e) => {
  e.preventDefault();   // stop default browser action
  e.stopPropagation();   // stop event bubbling up DOM
});
Event Propagation — Bubble & Capture
JavaScript
/*
  Event flow: 3 phases
  1. CAPTURE  — document → target   (top down)
  2. TARGET   — event is at target
  3. BUBBLE   — target → document   (bottom up, default)
*/

// By default listeners use BUBBLE phase
parent.addEventListener('click', () => console.log('parent bubble'));

// Pass true or {capture:true} to use CAPTURE phase
parent.addEventListener('click', () => console.log('parent capture'), true);

// Clicking child fires in order:
//   1. parent capture
//   2. child (target phase)
//   3. parent bubble

// Event Delegation — one handler for many children
const list = document.querySelector('#list');
list.addEventListener('click', (e) => {
  // e.target = the actual clicked element
  if (e.target.matches('li')) {
    console.log('item clicked:', e.target.textContent);
  }
});

Debounce & Throttle — Controlling Event Rate
JavaScript
// DEBOUNCE — wait until N ms after the LAST call
// Use for: search input, window resize
function debounce(fn, delay) {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
}

const debouncedSearch = debounce((q) => searchAPI(q), 300);
input.addEventListener('input', (e) => debouncedSearch(e.target.value));

// THROTTLE — fire at most once per N ms
// Use for: scroll, mousemove, animation frames
function throttle(fn, limit) {
  let last = 0;
  return (...args) => {
    const now = Date.now();
    if (now - last >= limit) {
      last = now;
      fn(...args);
    }
  };
}

const throttledScroll = throttle(onScroll, 100);
window.addEventListener('scroll', throttledScroll);

// RAF-based throttle (best for visual updates)
let rafId;
window.addEventListener('scroll', () => {
  cancelAnimationFrame(rafId);
  rafId = requestAnimationFrame(onScroll);
});

// EventEmitter pattern (pure JS)
class EventEmitter {
  #handlers = new Map();

  on(event, handler) {
    if (!this.#handlers.has(event))
      this.#handlers.set(event, []);
    this.#handlers.get(event).push(handler);
    return this;
  }

  emit(event, data) {
    this.#handlers.get(event)?.forEach(h => h(data));
    return this;
  }

  off(event, handler) {
    const handlers = this.#handlers.get(event);
    if (handlers) {
      this.#handlers.set(event, handlers.filter(h => h !== handler));
    }
    return this;
  }
}

Node.js Event Loop
Node.js uses a single-threaded, non-blocking event loop built on libuv. I/O operations (file reads, network calls) are offloaded to the OS; when complete, their callbacks are queued and executed by the event loop. This is why Node excels at I/O-heavy workloads despite being single-threaded.

Node.js EventEmitter
Node.js
import { EventEmitter } from 'node:events';

// Extend EventEmitter
class DataPipeline extends EventEmitter {
  process(items) {
    for (const item of items) {
      try {
        const result = this.transform(item);
        this.emit('data', result);         // emit event + payload
      } catch (err) {
        this.emit('error', err);           // 'error' is special in Node
      }
    }
    this.emit('end');
  }
  transform(x) { return x * 2; }
}

const pipeline = new DataPipeline();

pipeline.on('data',  (r)   => console.log('Result:', r));
pipeline.on('error', (err) => console.error('Error:', err));
pipeline.once('end', ()    => console.log('Done'));  // one-time listener

pipeline.process([1, 2, 3]);

// promisify — wrap EventEmitter in a Promise
function waitFor(emitter, event) {
  return new Promise((resolve, reject) => {
    emitter.once(event,   resolve);
    emitter.once('error', reject);
  });
}

await waitFor(pipeline, 'end');
console.log('Pipeline finished');

import { createServer } from 'node:http';

const server = createServer(async (req, res) => {
  const url = new URL(req.url, `http://${req.headers.host}`);

  if (req.method === 'POST' && url.pathname === '/api/data') {
    // Read body stream

    return;

Promise Combinators
JavaScript
const p1 = fetch('/api/user').then(r => r.json());
const p2 = fetch('/api/posts').then(r => r.json());

  if (r.status === 'fulfilled') use(r.value);
  else                            log(r.reason);
});

// race — first to settle (fulfilled OR rejected) wins
const winner = await Promise.race([p1, timeout(5000)]);

// any — first FULFILLED wins; rejects if ALL reject
const fastest = await Promise.any([mirror1, mirror2, mirror3]);

async function* paginate(url) {
  let nextUrl = url;
  while (nextUrl) {

Common Async Patterns
JavaScript
// Retry with exponential backoff
async function retry(fn, retries = 3, delay = 300) {
  for (let i = 0; i <= retries; i++) {
    try { return await fn(); }
    catch (err) {
      if (i === retries) throw err;
      await new Promise(r => setTimeout(r, delay * 2**i));
    }
  }
}

// Timeout wrapper
function withTimeout(promise, ms) {
  const timeout = new Promise((_, reject) =>
    setTimeout(() => reject(new Error('Timeout')), ms)
  );
  return Promise.race([promise, timeout]);
}

// Queue / rate limit parallel requests
async function batchProcess(items, fn, concurrency = 3) {
  const results = [];
  for (let i = 0; i < items.length; i += concurrency) {
    const batch = items.slice(i, i + concurrency);
    results.push(...await Promise.all(batch.map(fn)));
  }
  return results;
}

Reactive Programming
RxJS treats events as streams over time that can be composed, transformed, and combined using operators. An Observable is a lazy push-based collection — nothing happens until you subscribe().
Observable & Observer
RxJS
import { Observable, of, from, fromEvent, interval } from 'rxjs';

// Create observable manually
const obs$ = new Observable<number>(subscriber => {
  subscriber.next(1);    // push value
  subscriber.next(2);
  subscriber.next(3);
  subscriber.complete(); // stream ends
  // subscriber.error(err); — stream errors
  return () => cleanup(); // teardown on unsubscribe
});

// Subscribe (Observer object)
const sub = obs$.subscribe({
  next:     (v) => console.log('Value:', v),
  error:    (e) => console.error('Error:', e),
  complete: ()  => console.log('Done'),
});

sub.unsubscribe();  // stop receiving, run teardown

// Creation helpers
of(1, 2, 3)              // from discrete values
from([1, 2, 3])           // from array/promise/iterable
interval(1000)            // emit 0,1,2... every second
fromEvent(btn, 'click')   // DOM events as stream

Essential Operators (pipe)
RxJS
import {
  map, filter, take, skip, debounceTime, distinctUntilChanged,
  switchMap, mergeMap, concatMap, exhaustMap,
  catchError, retry, tap, shareReplay, combineLatest, withLatestFrom
} from 'rxjs/operators';

// Operators compose with pipe()
search$.pipe(
  debounceTime(300),            // wait 300ms after last value
  distinctUntilChanged(),        // skip if same as last value
  filter(q => q.length >= 2),  // only if length >= 2
  switchMap(q =>                  // cancel prev inner obs, subscribe new
    from(searchAPI(q)).pipe(
      catchError(() => of([]))    // don't kill stream on error
    )
  )
).subscribe(results => render(results));

// Flattening strategies (inner observable handling):
// mergeMap   — run all concurrently (unordered)
// concatMap  — queue sequentially (preserves order)
// switchMap  — cancel prev on new value (search/typeahead)
// exhaustMap — ignore new until current completes (login btn)

// Combining observables
combineLatest([user$, settings$])
  .subscribe(([user, settings]) => render(user, settings));

// ShareReplay — multicast + cache last N values for late subscribers
const shared$ = expensiveOp$.pipe(shareReplay(1));
Subjects — Observable + Observer
RxJS
import { Subject, BehaviorSubject, ReplaySubject } from 'rxjs';

// Subject — multicast, no initial value
const events$ = new Subject<string>();
events$.subscribe(v => console.log('A:', v));
events$.subscribe(v => console.log('B:', v));
events$.next('hello');  // both A and B receive 'hello'

// BehaviorSubject — has current value, emits to new subscribers
const count$ = new BehaviorSubject(0);
count$.next(1);
count$.subscribe(v => console.log(v));  // immediately gets 1
console.log(count$.value);               // synchronous read: 1

// ReplaySubject — replays N last values to new subscribers
const log$ = new ReplaySubject(3);  // buffer 3
log$.next('a'); log$.next('b'); log$.next('c'); log$.next('d');
log$.subscribe(v => console.log(v)); // gets b, c, d
