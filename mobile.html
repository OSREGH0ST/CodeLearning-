// guard — early exit pattern
func process(input: String?) {
    guard let input = input, !input.isEmpty else {
        print("no input")
        return
    }
    print(input)  // input is non-nil here
}

// switch — exhaustive, no fallthrough by default
switch score {
case 90...100:
    grade = "A"
case 70..<90:
    grade = "B"
default:
    grade = "F"
}

// Pattern matching in switch

case (0, 0): print("origin")
case (let x, 0): print("x-axis at \(x)")
case (_, _): print("other")
}

enum Shape {
    case circle(radius: Double)
    case rectangle(width: Double, height: Double)

    var area: Double {
        switch self {
        case .circle(let r): return .pi * r * r

        }
    }

// Struct vs Class: struct = value type, class = reference type
struct Point {
    var x, y: Double
    mutating func translate(dx: Double, dy: Double) {
        x += dx; y += dy   // must be mutating to change self
    }
}

// Protocol — like interface
protocol Drawable {
    func draw()
    var color: String { get }
}

// Protocol extension — default implementation
extension Drawable {
    func hide() { print("hidden") }
}

// Class with inheritance
class Animal {
    let name: String
    init(name: String) { self.name = name }
    func speak() -> String { "..." }
}

class Dog: Animal, Drawable {
    var color: String = "brown"
    override func speak() -> String { "Woof" }
    func draw() { print("🐕") }
}

// Generics + Protocols
func largest<T: Comparable>(_ a: [T]) -> T? {
    return a.max()
}

// @propertyWrapper
@propertyWrapper
struct Clamped {
    private var value: Int
    let range: ClosedRange<Int>
    var wrappedValue: Int {
        get { value }
        set { value = min(max(newValue, range.lowerBound), range.upperBound) }
    }
}

struct Config {
    @Clamped(range: 0...100) var volume: Int = 50
}

Coroutines (Async)
Kotlin
import kotlinx.coroutines.*

// suspend function — can be paused
suspend fun fetchData(): String {
    delay(1000)  // non-blocking wait
    return "data"
}

// launch — fire and forget
GlobalScope.launch {
    val result = fetchData()
    println(result)
}

// async / await — parallel tasks
val deferred1 = async { fetchData() }
val deferred2 = async { fetchData() }
val (r1, r2)  = deferred1.await() to deferred2.await()

// Flow — cold stream
fun numbersFlow(): Flow<Int> = flow {
    for (i in 1..5) {
        delay(100)
        emit(i)
    }
}

numbersFlow()
    .filter { it % 2 == 0 }
    .collect { println(it) }

Dart OOP
Dart
class Animal {
  String _name;       // private (library-private by _)
  late String sound;  // late = initialized later

  // Named constructor
  Animal(this._name, this.sound);
  Animal.silent(String name) : this(name, '...');

  String get name => _name;
  set name(String v) {
    if (v.isEmpty) throw ArgumentError('empty');
    _name = v;
  }

  String speak() => '$_name: $sound';
}

class Dog extends Animal {
  Dog(String name) : super(name, 'Woof');

  @override
  String speak() => '$name barks!';
}

// Mixin — reuse code without inheritance
mixin Swimmable {
  void swim() => print('swimming');
}

class Duck extends Animal with Swimmable {
  Duck() : super('Duck', 'Quack');
}
Flutter Widget Pattern
Dart/Flutter
import 'package:flutter/material.dart';

// Stateless widget — no mutable state
class Greeting extends StatelessWidget {
  final String name;
  const Greeting({super.key, required this.name});

  @override
  Widget build(BuildContext ctx) {
    return Text('Hello, $name');
  }
}

// Stateful widget — has mutable state
class Counter extends StatefulWidget {
  const Counter({super.key});
  @override
  State<Counter> createState() => _CounterState();
}

class _CounterState extends State<Counter> {
  int _count = 0;

  @override
  Widget build(BuildContext ctx) {
    return GestureDetector(
      onTap: () => setState(() => _count++),
      child: Text('Count: $_count'),
    );
  }
}
