Classes & OOP
Python
from dataclasses import dataclass, field
from abc import ABC, abstractmethod

class Animal:
    # Class variable (shared)
    count: int = 0

    def __init__(self, name: str, sound: str) -> None:
        self._name  = name       # protected (convention)
        self.__sound = sound     # private (name-mangled)
        Animal.count += 1

    @property
    def name(self) -> str:
        return self._name

    @name.setter
    def name(self, v: str) -> None:
        if not v: raise ValueError("empty")
        self._name = v

    def speak(self) -> str:
        return f"{self._name}: {self.__sound}"

    @classmethod
    def from_dict(cls, d: dict) -> "Animal":
        return cls(d["name"], d["sound"])

    @staticmethod
    def validate(name: str) -> bool:
        return bool(name.strip())

    def __repr__(self) -> str:
        return f"Animal({self._name!r})"

# Inheritance
class Dog(Animal):
    def __init__(self, name: str, breed: str):
        super().__init__(name, "Woof")
        self.breed = breed

    def speak(self) -> str:  # override
        return f"{self._name} barks!"

# Abstract base class
class Shape(ABC):
    @abstractmethod
    def area(self) -> float: ...

# Dataclass — auto __init__, __repr__, __eq__
@dataclass
class Point:
    x: float
    y: float
    tags: list = field(default_factory=list)
